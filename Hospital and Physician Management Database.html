    if personnel:
        doctor['personnelID'] = personnel['_id']

    # Relationship for prescribing doctor in Prescriptions
    prescribing_doctor_id = doctor['doctorID']
    prescriptions_data_for_doctor = prescriptions_collection.find({'prescribingDoctorID': prescribing_doctor_id})
    prescriptions_ids = [prescription['_id'] for prescription in prescriptions_data_for_doctor]
    doctors_collection.update_one({'doctorID': prescribing_doctor_id}, {'$set': {'personnelID': doctor.get('personnelID', None), 'prescriptions': prescriptions_ids}})

    # Relationship for testing doctor and ordering doctor in Diagnostic Tests

    diagnostic_tests_data_for_doctor = diagnostic_tests_collection.find({
        '$or': [
            {'testingDoctorID': doctor_id},
            {'orderingDoctorID': doctor_id}
        ]
    })
    diagnostic_tests_ids = [diagnostic_test['_id'] for diagnostic_test in diagnostic_tests_data_for_doctor]
